CREATE TABLE tTask (
    ID serial PRIMARY KEY,
    description varchar (512) NOT NULL,
    taskStatusID integer REFERENCES tTaskStatus NOT NULL,
    isDeleted boolean DEFAULT false NOT NULL,
    createdMoment timestamp DEFAULT CURRENT_TIMESTAMP NOT NULL,
    completedMoment timestamp,
    changedMoment timestamp
);
